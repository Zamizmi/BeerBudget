30      /**
31       * Is the first method to run from the main-class. Is a text-based User
32       * Interface.
33       */
34      public void run() {
35          System.out.println("Welcome to the Beer Budget!");
36          System.out.println("Please start by telling your current balance");
37          Double balance = Double.parseDouble(reader.nextLine());
38          logic.setBalance(balance);
39          System.out.println("Please tell me your target sum you wish to save");
40          Double target = Double.parseDouble(reader.nextLine());
41          logic.setTarget(target);
42          logic.setCurrentDate(askForDate());
43          listCommands();
44          commands();
45      }

56      /**
57       * Returns all the commands available for the text-based UI.
58       * @return all commands.
59       */
60      public ArrayList<String> listCommands() {
61          ArrayList commands = new ArrayList<>();
62          commands.add("list - Prints all the commands");
63          commands.add("moneyflow - Add new expense or income");
64          commands.add("balance - Print current balance");
65          commands.add("budget - View the current budget");
66          commands.add("incomes - Prints all added incomes");
67          commands.add("expenses - Prints all added expenses");
68          commands.add("target - Prints current target and how long you will have save for it at current budget");
69          commands.add("First of May -  Prints the number of beers you will get at the next First of May!");
70          return commands;
71      }

73      /**
74       * Is running while app is running in text-based. Asks for input and calls
75       * the right methods after receiving commands.
76       */
77      public void commands() {
78          while (true) {
79              System.out.println("Please give one of the commands to proceed.");
80              String command = reader.nextLine();
81              if (command.equals("list")) {
82                  listCommands();
83              } else if (command.equals("moneyflow")) {
84                  askForMoneyFlow();
85              } else if (command.equals("balance")) {
86                  currentBalance();
87              } else if (command.equals("budget")) {
88                  currentBudget();
89              } else if (command.equals("incomes")) {
90                  listIncomes();
91              } else if (command.equals("expenses")) {
92                  listExpenses();
93              } else if (command.equals("target")) {
94                  printTarget();
95              } else if (command.equals("First of May")) {
96                  firstOfMay();
97              } else if (command.equals("exit")) {
98                  System.out.println("SEE YA MONEY MASTER!");
99                  break;
100             } else {
101                 System.out.println("Command was invalid!");
102             }
103         }
104     }

186     /**
187      * Asks for input and user creates new moneyflow. The new Moneyflow is added
188      * to the budget automatically.
189      */
190     public void askForMoneyFlow() {
191         while (true) {
192             System.out.println("Write 'i' for new Income, 'e' for new expense, exit to stop");
193             String decision = reader.nextLine();
194             if (decision.equals("i")) {
195                 newIncome();
196             } else if (decision.equals("e")) {
197                 newExpense();
198             } else if (decision.equals("exit")) {
199                 break;
200             } else {
201                 System.out.println("Say that again!");
202             }
203         }
204     }

206     /**
207      * Asks for input from the User to create expense.
208      */
209     public void newExpense() {
210         System.out.println("Give a name for the expense");
211         String name = reader.nextLine();
212         System.out.println("How much money will it cost?");
213         double amount = Double.parseDouble(reader.nextLine());
214         boolean monthly = false;
215         while (true) {
216             System.out.println("Will it be monthly(m) or once(o)?");
217             String decision = reader.nextLine();
218             if (decision.equals("m")) {
219                 monthly = true;
220                 break;
221             } else if (decision.equals("o")) {
222                 monthly = false;
223                 break;
224             } else {
225                 System.out.println("You chose expense to be paid once");
226                 break;
227             }
228         }
229         Date expiration = askForDate();
230         this.logic.newExpense(name, amount, new Date(), monthly);
231     }

245             String decision = reader.nextLine();
246             if (decision.equals("m")) {
247                 monthly = true;
248                 break;
249             } else if (decision.equals("o")) {
250                 monthly = false;

252             } else {
253                 System.out.println("Input is invalid, try again!");

255         }
256         Date expiration = askForDate();
257         this.logic.newIncome(name, amount, expiration, monthly);
258     }
