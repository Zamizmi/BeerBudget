46      private Scene startScene() {
47          Label currentBalance = new Label("Current Balance: ");
48          TextField balanceInput = new TextField();
49          Label targetText = new Label("Target: ");
50          TextField targetInput = new TextField();
51  
52          Button continueButton = new Button("Push to Continue");
53          continueButton.setOnAction((event) -> {
54              this.window.setScene(basicMenu());
55          });
56  
57          GridPane startView = new GridPane();
58          startView.add(currentBalance, 0, 0);
59          startView.add(balanceInput, 1, 0);
60          startView.add(targetText, 0, 1);
61          startView.add(targetInput, 1, 1);
62          startView.add(continueButton, 1, 2);
63          startView.setHgap(10);
64          startView.setVgap(10);
65          startView.setPadding(new Insets(10, 10, 10, 10));
66  
67          Scene scene = new Scene(startView);
68          return scene;
69          //throw new UnsupportedOperationException("Not supported yet."); //To change body of generated methods, choose Tools | Templates.
70      }
