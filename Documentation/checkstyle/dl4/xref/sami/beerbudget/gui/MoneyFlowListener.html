20   * @author saklindq

22  public class MoneyFlowListener implements ActionListener {
23  
24      private BudgetLogic bl;
25      private JFrame frame;
26      private JTextField name;
27      private JTextField amount;
28      private JTextField date;
29      private boolean monthly;
30      private Boolean expense;
31  
32      public MoneyFlowListener(BudgetLogic bl, JFrame jf, JTextField nameField, JTextField amountField, JTextField dateField, boolean monthly, boolean expense) {
33          this.bl = bl;
34          this.frame = jf;
35          this.name = nameField;
36          this.amount = amountField;
37          this.date = dateField;
38          this.monthly = monthly;
39          this.expense = expense;
40      }
41      
42      //TODO
43      /**
44       * When activated, creates new moneyflow.
45       * @param ae activation.
46       */
47      @Override
48      public void actionPerformed(ActionEvent ae) {
49          String command = ae.getActionCommand();
50          try {
51              double amount = Double.parseDouble(this.amount.getText());
52              String name = this.name.getText();
53              Date dateType = DateLogic.stringToDate(this.date.getText());
54              //System.out.println(amount);
55              if (command.equals("Lisaa")) {
56                  if (expense) {
57                      bl.newExpense(name, amount, dateType, monthly);
58                      this.frame.dispose();
59                  } else {
60                      bl.newIncome(name, amount, dateType, monthly);
61                      this.frame.dispose();
62                  }
63              }
64          } catch (Exception e) {
65              this.frame.dispose();
66          }
67  
68      }
69  
70  }
